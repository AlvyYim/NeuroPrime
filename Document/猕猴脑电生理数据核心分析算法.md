以下是一个完整的、可运行的 Python 代码示例，模拟了猕猴电生理实验数据（含 LFP 和 spike），并实现了从预处理到解码分析的核心流程。代码使用合成数据，因此无需真实数据即可运行，并演示了每个关键步骤的算法实现。

```python
"""
猕猴电生理数据分析完整流程示例 (Python)
包含：数据生成、预处理、Spike Sorting、LFP分析、行为对齐、调谐曲线、解码分析等。
依赖库：numpy, scipy, matplotlib, sklearn, mne (可选，用于滤波)
"""

import numpy as np
import matplotlib.pyplot as plt
from scipy import signal
from scipy.stats import zscore
from sklearn.decomposition import PCA
from sklearn.cluster import KMeans
from sklearn.discriminant_analysis import LinearDiscriminantAnalysis
from sklearn.metrics import roc_curve, auc
import warnings
warnings.filterwarnings('ignore')

# ----------------------------
# 1. 模拟数据生成
# ----------------------------
def generate_simulated_data(duration=10.0, fs=30000, n_channels=4, spike_rate=5.0):
    """
    生成模拟的宽频电生理数据（含LFP和Spike）。
    - duration: 时长(秒)
    - fs: 采样率(Hz)
    - n_channels: 通道数
    - spike_rate: 平均发放率(Hz)
    返回: raw_data (n_channels, n_samples), spike_times列表, 模拟行为事件
    """
    n_samples = int(duration * fs)
    t = np.linspace(0, duration, n_samples)
    
    # 生成 LFP 成分 (0-100 Hz 的混合正弦波 + 1/f 噪声)
    lfp = np.zeros((n_channels, n_samples))
    for ch in range(n_channels):
        lfp[ch] = (0.5 * np.sin(2*np.pi*8*t) +           # theta
                   0.3 * np.sin(2*np.pi*20*t) +          # beta
                   0.2 * np.sin(2*np.pi*60*t))           # gamma 低频
        # 添加1/f噪声
        noise = np.cumsum(np.random.randn(n_samples)) * 0.01
        lfp[ch] += noise
        # 添加50Hz工频干扰
        lfp[ch] += 0.1 * np.sin(2*np.pi*50*t)
    
    # 生成 Spikes (每个通道独立的随机发放)
    spike_times_per_ch = []
    spike_waveform = signal.wavelets.ricker(50, 5)  # 模拟尖峰波形 (50点)
    for ch in range(n_channels):
        n_spikes = int(spike_rate * duration)
        spike_times = np.sort(np.random.uniform(0, duration, n_spikes))
        spike_times_per_ch.append(spike_times)
    
    # 将Spike叠加到数据中
    raw_data = lfp.copy()
    for ch, spike_times in enumerate(spike_times_per_ch):
        for st in spike_times:
            idx = int(st * fs)
            if idx + len(spike_waveform) < n_samples:
                raw_data[ch, idx:idx+len(spike_waveform)] += spike_waveform * 100  # 幅度放大
    
    # 模拟行为事件: 每2秒一个视觉刺激 (方向随机)
    n_trials = int(duration / 2.0)
    stim_times = np.arange(1.0, duration, 2.0)   # 刺激开始时间
    stim_directions = np.random.choice([-1, 1], n_trials)  # -1:左, 1:右 (用于调谐)
    
    return raw_data, fs, spike_times_per_ch, stim_times, stim_directions

# ----------------------------
# 2. 预处理: 滤波 + 重参考
# ----------------------------
def preprocess_data(raw_data, fs, lfp_lowpass=250, spike_band=[300, 6000]):
    """
    预处理：分离LFP和Spike信号，重参考(CAR)
    返回: lfp_data (低通滤波), spike_data (带通滤波), 以及CAR重参考后的版本
    """
    # 设计滤波器
    nyq = fs / 2
    # LFP 低通
    b_lfp, a_lfp = signal.butter(4, lfp_lowpass/nyq, 'low')
    lfp_filtered = signal.filtfilt(b_lfp, a_lfp, raw_data, axis=1)
    
    # Spike 带通
    b_spike, a_spike = signal.butter(4, [spike_band[0]/nyq, spike_band[1]/nyq], 'bandpass')
    spike_filtered = signal.filtfilt(b_spike, a_spike, raw_data, axis=1)
    
    # 共平均参考 (CAR) - 对 Spike 信号做
    mean_spike = np.mean(spike_filtered, axis=0)
    spike_car = spike_filtered - mean_spike
    
    return lfp_filtered, spike_car

# ----------------------------
# 3. Spike Detection & Sorting (简单版)
# ----------------------------
def detect_spikes(spike_signal, fs, threshold_sd=4):
    """
    阈值法检测Spike (基于标准差)
    返回: 每个通道的spike时间(秒)和波形列表
    """
    spike_times_list = []
    waveforms_list = []
    pre_samples = int(0.001 * fs)   # 1ms 前
    post_samples = int(0.002 * fs)  # 2ms 后
    for ch in range(spike_signal.shape[0]):
        data = spike_signal[ch, :]
        threshold = threshold_sd * np.std(data)
        # 寻找过阈值的峰值
        peaks, _ = signal.find_peaks(data, height=threshold, distance=int(0.001*fs))  # 间隔>1ms
        spike_times = peaks / fs
        # 提取波形
        waveforms = []
        valid_peaks = []
        for p in peaks:
            if p > pre_samples and p + post_samples < len(data):
                waveforms.append(data[p-pre_samples:p+post_samples])
                valid_peaks.append(p)
        spike_times = np.array(valid_peaks) / fs
        spike_times_list.append(spike_times)
        waveforms_list.append(np.array(waveforms))
    return spike_times_list, waveforms_list

def spike_sorting(waveforms, n_clusters=2):
    """
    基于PCA+KMeans的简单聚类
    返回: 聚类标签
    """
    if len(waveforms) == 0:
        return np.array([])
    # PCA 降维
    pca = PCA(n_components=3)
    features = pca.fit_transform(waveforms)
    # KMeans 聚类
    kmeans = KMeans(n_clusters=n_clusters, random_state=0, n_init=10)
    labels = kmeans.fit_predict(features)
    return labels

# ----------------------------
# 4. LFP 分析: 功率谱 & 时频图
# ----------------------------
def compute_power_spectrum(lfp_data, fs, f_range=(0, 100)):
    """计算多通道平均功率谱 (Welch)"""
    nperseg = int(1 * fs)  # 1秒窗口
    freqs, psd = signal.welch(lfp_data, fs, nperseg=nperseg, axis=1)
    mean_psd = np.mean(psd, axis=0)
    mask = (freqs >= f_range[0]) & (freqs <= f_range[1])
    return freqs[mask], mean_psd[mask]

def compute_spectrogram(lfp_signal, fs, window=0.5, step=0.05):
    """计算单通道时频谱 (STFT)"""
    nperseg = int(window * fs)
    noverlap = nperseg - int(step * fs)
    f, t, Sxx = signal.spectrogram(lfp_signal, fs, nperseg=nperseg, noverlap=noverlap)
    return f, t, 10*np.log10(Sxx + 1e-12)  # dB

# ----------------------------
# 5. 行为对齐: PSTH & 栅格图
# ----------------------------
def align_spikes_to_events(spike_times, event_times, t_before=0.5, t_after=1.0, bin_size=0.02):
    """
    将spike时间对齐到事件，返回PSTH和栅格数据
    """
    n_events = len(event_times)
    bins = np.arange(-t_before, t_after + bin_size, bin_size)
    psth = np.zeros(len(bins)-1)
    raster = []
    for et in event_times:
        spikes_rel = spike_times - et
        spikes_rel = spikes_rel[(spikes_rel >= -t_before) & (spikes_rel <= t_after)]
        raster.append(spikes_rel)
        hist, _ = np.histogram(spikes_rel, bins=bins)
        psth += hist
    psth = psth / (n_events * bin_size)  # 发放率 (Hz)
    return bins, psth, raster

# ----------------------------
# 6. 调谐曲线 & ROC
# ----------------------------
def compute_tuning_curve(spike_counts, stimulus_values, n_bins=10):
    """计算调谐曲线 (假设刺激值连续，如方向角度)"""
    bins = np.linspace(min(stimulus_values), max(stimulus_values), n_bins+1)
    bin_centers = (bins[:-1] + bins[1:]) / 2
    mean_rates = []
    for i in range(n_bins):
        idx = (stimulus_values >= bins[i]) & (stimulus_values < bins[i+1])
        if np.sum(idx) > 0:
            mean_rates.append(np.mean(spike_counts[idx]))
        else:
            mean_rates.append(0)
    return bin_centers, np.array(mean_rates)

def roc_analysis(spike_counts, labels):
    """计算ROC和AUC (二分类)"""
    fpr, tpr, _ = roc_curve(labels, spike_counts)
    roc_auc = auc(fpr, tpr)
    return fpr, tpr, roc_auc

# ----------------------------
# 7. 解码分析: LDA
# ----------------------------
def decode_with_lda(feature_matrix, labels, cv_folds=5):
    """使用LDA进行解码，返回交叉验证准确率"""
    from sklearn.model_selection import cross_val_score
    lda = LinearDiscriminantAnalysis()
    scores = cross_val_score(lda, feature_matrix, labels, cv=cv_folds)
    return np.mean(scores)

# ----------------------------
# 主流程: 运行完整分析
# ----------------------------
def main():
    print("生成模拟数据...")
    raw_data, fs, true_spikes, stim_times, stim_directions = generate_simulated_data(
        duration=10.0, fs=30000, n_channels=4, spike_rate=8.0)
    print(f"数据形状: {raw_data.shape}, 采样率: {fs} Hz")
    
    # 预处理
    print("预处理: 滤波 + 重参考...")
    lfp_data, spike_car = preprocess_data(raw_data, fs)
    print(f"LFP形状: {lfp_data.shape}, Spike-CAR形状: {spike_car.shape}")
    
    # Spike 检测与排序
    print("Spike 检测...")
    detected_spikes, waveforms = detect_spikes(spike_car, fs, threshold_sd=4)
    # 对第一个通道进行聚类 (演示)
    ch_idx = 0
    if len(waveforms[ch_idx]) > 0:
        labels = spike_sorting(waveforms[ch_idx], n_clusters=2)
        print(f"通道 {ch_idx} 检测到 {len(labels)} 个Spike, 聚类为 {np.unique(labels)}")
    else:
        print("未检测到足够Spike")
        labels = []
    
    # LFP 功率谱分析
    print("LFP 功率谱...")
    freqs, psd = compute_power_spectrum(lfp_data, fs, f_range=(0, 100))
    plt.figure(figsize=(12, 10))
    plt.subplot(2,2,1)
    plt.plot(freqs, psd)
    plt.xlabel('Frequency (Hz)')
    plt.ylabel('Power')
    plt.title('平均功率谱 (0-100 Hz)')
    plt.grid(True)
    
    # LFP 时频图 (取第一通道)
    plt.subplot(2,2,2)
    f, t, Sxx = compute_spectrogram(lfp_data[0], fs, window=0.5, step=0.05)
    plt.pcolormesh(t, f, Sxx, shading='gouraud', cmap='jet')
    plt.ylabel('Frequency (Hz)')
    plt.xlabel('Time (s)')
    plt.title('LFP时频图 (通道0)')
    plt.colorbar(label='PSD (dB)')
    
    # 行为对齐: PSTH & 栅格图 (以刺激事件对齐)
    print("行为对齐分析...")
    # 选取一个通道的spike时间
    spike_times_ch0 = detected_spikes[ch_idx]
    bins, psth, raster = align_spikes_to_events(spike_times_ch0, stim_times, 
                                                 t_before=0.5, t_after=1.0, bin_size=0.02)
    plt.subplot(2,2,3)
    # 栅格图 (只显示前20个trial避免过密)
    for i, spikes in enumerate(raster[:20]):
        plt.scatter(spikes, np.ones_like(spikes)*i, s=1, c='k')
    plt.xlabel('Time relative to stimulus (s)')
    plt.ylabel('Trial #')
    plt.title('栅格图 (前20 trials)')
    plt.xlim(-0.5, 1.0)
    
    plt.subplot(2,2,4)
    bin_centers = (bins[:-1] + bins[1:]) / 2
    plt.plot(bin_centers, psth)
    plt.xlabel('Time (s)')
    plt.ylabel('Firing rate (Hz)')
    plt.title('PSTH')
    plt.grid(True)
    plt.tight_layout()
    
    # 调谐曲线: 假设刺激方向(-1,1)影响发放率，模拟一个调谐响应
    # 这里使用真实刺激方向，并计算每个trial的平均发放率
    trial_spike_counts = []
    for i, stim_time in enumerate(stim_times):
        # 统计刺激后0-0.5秒内的spike数
        spikes_in_window = spike_times_ch0[(spike_times_ch0 >= stim_time) & (spike_times_ch0 <= stim_time+0.5)]
        trial_spike_counts.append(len(spikes_in_window))
    trial_spike_counts = np.array(trial_spike_counts)
    
    # 按方向分组
    plt.figure(figsize=(12, 4))
    plt.subplot(1,3,1)
    bin_centers, tuning = compute_tuning_curve(trial_spike_counts, stim_directions, n_bins=2)
    plt.bar(bin_centers, tuning, width=0.8)
    plt.xlabel('Stimulus direction')
    plt.xticks([-1, 1], ['Left', 'Right'])
    plt.ylabel('Mean spike count (0-0.5s)')
    plt.title('调谐曲线')
    
    # ROC 分析: 区分左右刺激
    plt.subplot(1,3,2)
    fpr, tpr, roc_auc = roc_analysis(trial_spike_counts, (stim_directions+1)//2)  # 转为0/1
    plt.plot(fpr, tpr, label=f'AUC = {roc_auc:.3f}')
    plt.plot([0,1], [0,1], 'k--')
    plt.xlabel('False Positive Rate')
    plt.ylabel('True Positive Rate')
    plt.title('ROC曲线')
    plt.legend()
    
    # 解码分析: 使用多通道群体活动 (每个trial各通道的发放率)
    # 构造特征矩阵: trials x channels
    feature_matrix = []
    for ch in range(raw_data.shape[0]):
        spike_times_ch = detected_spikes[ch]
        counts = []
        for stim_time in stim_times:
            cnt = np.sum((spike_times_ch >= stim_time) & (spike_times_ch <= stim_time+0.5))
            counts.append(cnt)
        feature_matrix.append(counts)
    feature_matrix = np.array(feature_matrix).T  # (n_trials, n_channels)
    labels_binary = (stim_directions + 1) // 2
    acc = decode_with_lda(feature_matrix, labels_binary)
    plt.subplot(1,3,3)
    plt.bar(['Decoding Accuracy'], [acc], yerr=[np.std([acc])], capsize=10)
    plt.ylim(0, 1)
    plt.ylabel('Accuracy')
    plt.title(f'LDA Cross-Validation (mean={acc:.3f})')
    plt.tight_layout()
    
    plt.show()
    print("分析完成。")

if __name__ == "__main__":
    main()
```

## 代码说明

1. **模拟数据生成**  
   `generate_simulated_data` 产生多通道宽频信号：LFP 为低频正弦波+噪声，Spike 为叠加的 Ricker 小波，并生成随机的视觉刺激事件（时间、方向）。

2. **预处理**  
   - 使用 Butterworth 滤波器分离 LFP（低通 250 Hz）和 Spike（带通 300-6000 Hz）。  
   - 对 Spike 信号执行共平均参考（CAR）。

3. **Spike 检测与排序**  
   - 阈值检测：基于标准差倍数（默认 4）寻找峰值。  
   - 聚类：对波形提取 PCA 前 3 个成分，再用 KMeans 分为 2 类（模拟多单元）。

4. **LFP 分析**  
   - 功率谱：Welch 方法，计算 0-100 Hz 平均谱。  
   - 时频图：短时傅里叶变换（STFT），显示能量变化。

5. **行为对齐**  
   - 栅格图与 PSTH：将 Spike 时间对齐到刺激开始时间，计算发放率直方图。

6. **调谐曲线与 ROC**  
   - 计算不同刺激条件下（左/右方向）的发放率，绘制调谐柱状图。  
   - ROC 分析评估单个神经元对左右刺激的区分能力。

7. **解码分析**  
   - 构建每个 trial 的多通道发放计数特征矩阵。  
   - 使用 LDA 进行 5 折交叉验证，输出平均解码准确率。
